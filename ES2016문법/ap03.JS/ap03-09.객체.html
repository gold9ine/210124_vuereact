객체 리터럴( {} ) 을 이용한 객체 선언 var foo = { name : 'foo', run : function () { return this.name; } }; 생성자 함수를 이용한 객체 선언. 상속 구현 시 사용 function Goo(name, age) { this.name = name; this.run = function () { return this.name; }; }; var goo = new Goo('goo', 31); ※생성자 함수란 첫 글자를 대문자로 시작하는 함수 // 리터럴을 이용한 객체 생성 var foo = { name : 'foo', age : 30, array : [1,4,5,5], run : function () { return this.name; } }; // CRUD : R. 읽기- 마침표 표기법 console.log( foo.name ); // CRUD : R. 읽기- 대괄호 표기법 console.log( foo['age'] ); console.log( foo.run() ); // CRUD : C. 추가 foo.gender = 'man'; foo['nickname'] = 'babo'; console.log( foo.gender ); console.log( foo.nickname ); /// CRUD : U. 수정 foo.name = 'bar'; foo['age'] = 50; console.log( foo.name ); console.log( foo.age ); // CRUD : D. 삭제 delete foo.age; delete foo.gender; console.log( foo.age ); console.log( foo.gender ); // 프로퍼티 수정 name 을 'bar' 로 변경 age 를 50 으로 변경 name과 age를 출력하시오 // 프로퍼티 추가 gender 프로퍼티 추가 nickname 프로퍼티 추가 gender 와 nickname 을 출력하시오 // 프로퍼티 삭제 nickname 프로퍼티 삭제 gender 프로퍼티 삭제 gender 와 nickname 을 출력하시오
